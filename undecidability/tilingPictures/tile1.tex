\newcommand{\tile}[4]
{
 \begin{tikzpicture}
\foreach \x in {0}
\foreach \y in {0}
{
\draw (\x, \y)    rectangle ++(2,2);
};
\draw  (1,1.7) node{\protect{$#1$}};  %% north
\draw (1.6,1) node{\protect{$#4$}}; %% east
\draw  (1,.3) node{\protect{$#2$}};  %% south
\draw (.4,1) node{\protect{$#3$}};%% west
\end{tikzpicture}
}
